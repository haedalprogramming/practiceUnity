    {
        controls = new PlayerControls();
        controls.Player.Move.performed += ctx => move = ctx.ReadValue<Vector2>();

    void OnEnable() => controls.Enable();
    void OnDisable() => controls.Disable();

    void Update()
    {
        transform.Translate(move.x * speed * Time.deltaTime, 0, 0);
    }

    void Jump()
    {
        Debug.Log("Jumped!");
    }
}
